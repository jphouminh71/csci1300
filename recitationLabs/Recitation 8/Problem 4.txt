int getLinesFromFile(string file_name, string myArray[], int size)
{
    ifstream file;
    
    file.open(file_name);
    
    if(file.is_open())
    {
        int count = 0;
        string lines;
        while (getline(file,lines))
        {
            if(lines.empty() == false)
            {
                myArray[count] = lines;
                count = count + 1;
            }
        }
        file.close();
        return count;
    }
    else
    {
        return -1;
    }
    
    file.close();
}
